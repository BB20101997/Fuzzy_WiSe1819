%% OP2 = kK^l_y   = y* min(a,b) + (1-y)(a+b)/2
%% OP3 = kK^r_y   = y* max(a,b) + (1-y)(a+b)/2 

%% alg_p(a,b) = ab     t-Norm
%% alg_s(a,b) = a+b-ab s-Norm

$A_0(a,b)    = [alg_p(a,b)]^{1-0} [alg_s(a,b)]^0 = alg_p(a,b) = ab$

$A_1(a,b)    = [alg_p(a,b)]^{1-1} [alg_s(a,b)]^1 = alg_s(a,b) = a+b-ab$

$kK^l_0(a,b) = 0 \cdot min(a,b) + (1-0)\frac{a+b}{2}          = \frac{a+b}{2}$

$kK^l_1(a,b) = 1 \cdot min(a,b) + (1-1)\frac{a+b}{2}          = min(a,b)$

$kK^r_0(a,b) = 0 \cdot max(a,b) + (1-0)\frac{a+b}{2}          = \frac{a+b}{2}$

$kK^r_1(a,b) = 1 \cdot max(a,b) + (1-1)\frac{a+b}{2}          = max(a,b)$

$K_0(a,b)    = [min(a,b)^{1-0}] [max(a,b)]^0                  = min(a,b)$

$K_1(a,b)    = [min(a,b)^{1-1}] [max(a,b)]^1                  = max(a,b)$
